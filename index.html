const SHEET_ID = "10NrnsbJPetNy4up8Ujj4IQJvq38CEkAlX9WauZ8gtto"; function doGet() { return ContentService .createTextOutput( JSON.stringify(getPendientes()) ) .setMimeType(ContentService.MimeType.JSON); } function getPendientes(){ const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName("Pendientes"); const data = sheet.getDataRange().getValues(); data.shift(); return data.map((r,i)=>({ fila:i+2, nombre:r[0], logo:r[1], stream:r[2], ciudad:r[3], pais:r[4], descripcion:r[5], web:r[6], whatsapp:r[7], facebook:r[8], instagram:r[9], url:r[10] })); } function doPost(e){ const data = JSON.parse(e.postData.contents); if(data.action==="aprobar"){ aprobarRadio(data.fila); } if(data.action==="eliminar"){ eliminarRadio(data.fila); } return ContentService .createTextOutput( JSON.stringify({ status:"ok" }) ) .setMimeType(ContentService.MimeType.JSON); } function aprobarRadio(fila){ const ss = SpreadsheetApp.openById(SHEET_ID); const pendientes = ss.getSheetByName("Pendientes"); const aprobadas = ss.getSheetByName("Aprobadas"); const data = pendientes .getRange(fila,1,1,11) .getValues()[0]; aprobadas.appendRow(data); pendientes.deleteRow(fila); } function eliminarRadio(fila){ const ss = SpreadsheetApp.openById(SHEET_ID); const pendientes = ss.getSheetByName("Pendientes"); pendientes.deleteRow(fila); }
